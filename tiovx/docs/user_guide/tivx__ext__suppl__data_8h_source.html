53 /* \brief Set supplementary data for an object descriptor from a user data object object descriptor by copying bytes from one
54 * user data object to another, and then extend the data in the user object by copying bytes from another place.
55 * \details The maximum number of bytes to copy is given by num_bytes.
56 * \details Data is first copied from the source object up to its valid size (given by the attribute VX_USER_DATA_OBJECT_VALID_SIZE),
57 * and then bytes are copied from the user_data, starting an equivalent distance in, until num_bytes have been copied.
58 * Thus the first part of user_data will be ignored if the valid size of the data in the source is greater than zero.
59 * If the copy operation reaches num_bytes before the bytes in source have been all copied, the operation stops without copying
60 * any data from user_data.
61 * If user_data is NULL, then it will copy a number of bytes from source to the destination supplementary data object, given
62 * by the minimum of num_bytes and the valid size.
63 * If source is NULL, then it will copy num_bytes from user_data to the destination supplementary data object and set the valid
64 * size to num_bytes.
65 * If num_bytes is greater than the VX_USER_DATA_OBJECT_SIZE attribute of the supplementary data then the function will fail and
66 * return VX_ERROR_INVALID_VALUE.
67 * If both user_data and source are NULL, the function will fail and return VX_ERROR_INVALID_PARAMETERS
68 * After the operation, the valid_mem_size of the destination supplementary data is set to the total number of bytes copied.
69  * \param [in] destination The object descriptor to be queried.
70  * \param [in] source Source of the data to be copied first (vx_user_data_object holding a some_user_data_t) or NULL
71  * \param [in] user_data Pointer to data object to be copy rest of bytes (some_user_data_t), or NULL
72  * \param [in] num_bytes Total number of bytes to copy, from source and user_data combined
73  * \return a status code:
74  * \retval VX_SUCCESS Data copied successfully
75  * \retval VX_FAILURE There is no supplementary data in destination
76  * \retval VX_ERROR_INVALID_VALUE num_bytes is greater than mem_size of source
77  * \retval VX_ERROR_INVALID_TYPE source type does not match destination supplementary data type
78  * \retval VX_ERROR_INVALID REFERENCE destination is NULL
79  * \retval VX_ERROR_INVALID_PARAMETERS source and user_Data are both NULL
80  */
81 VX_API_ENTRY vx_status tivxExtendSupplementaryDataObjDesc(const tivx_obj_desc_t * destination, const tivx_obj_desc_user_data_object_t * source, const void *user_data, vx_uint32 num_bytes);
